                {#2/#3/}%
                }
}

  {% new group to make changes to \count0 local
  \ifnum\mychappagenum>0
   \applynumberofpages
  \fi
  \count0=\value{page}
  \advance\count0 by #1
  \xdef\mychappagenum{\the\count0}
  \message{^^JChapter should end at page \mychappagenum^^J}
  }
}
