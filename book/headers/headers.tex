\newcommand{\ssainput}[2]{%
\message{Current ssadir set to #1 (file is #2)}
  \xdef\currentssadir{#1}%
  \input{\ssaroot{#1/#2}}%
}
\newcommand{\ssainputelse}[2]{
  \IfFileExists{\ssaroot{\currentssadir/#1}}%
  {\input{\ssaroot{\currentssadir/#1}}}%
  {#2}%
}
\newcommand{\ssausepackage}[2][]{%
  \usepackage[#1]{\ssaroot{packages/#2}}
}

\ssainput{headers}{packages.tex}
\ssainput{headers}{style.tex}
\ssainput{headers}{macros.tex}
